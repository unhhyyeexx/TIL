import sys

def circle(arr, start, row, column, count,K): #한바퀴 도는 함수
    lastrow = start + row -1
    lastcol = start + column -1
    #첫행에서 마지막행까지 내려가기
    for r in range(start, lastrow+1): 
        arr[r][start] = count
        count += 1
        if count-1 == K:
            return count-1
    #마지막행 첫열은 채워져 있음, 마지막행 두번째열부터 마지막행 마지막 열까지 ->방향   
    for c in range(start+1, lastcol+1):
        arr[lastrow][c] = count
        count += 1
        if count-1 == K:
            return count-1
    #마지막열, 마지막에서 두번째 행부터 첫번째 행까지 올라오기
    for r in range(lastrow -1, start-1,-1):
        arr[r][lastcol] = count
        count += 1
        if count-1 == K:
            return count-1
    #첫번째 행, 마지막에서 두번째 열부터 앞에서 두번째 열까지 <-방향  (첫행첫열 처음에 채움)
    for c in range(lastcol-1, start, -1):
        arr[start][c] = count
        count += 1
        if count-1 == K:
            return count -1
    return count

C, R = map(int, sys.stdin.readline().split())
K = int(sys.stdin.readline())

if R == 0:
    print(0) if K>C else print(f'{K} 1')
else:
    if K > C*R:
        print(0)
    else: 
        seat = [[ 0 for _ in range(C)] for _ in range(R)]
        count = 1
        st = 0
        if K == 1:
            print('1 1')
        while count < K:  #K 찾을때까지만
            count = circle(seat, st, R, C, count, K)
            C -= 2 #한바퀴 돌 때마다 행열 두줄씩 줄어듦
            R -= 2
            st += 1 

        for i in range(len(seat)):
            for j in range(len(seat[i])):
                if seat[i][j] == K:
                    print(f'{j+1} {i+1}')
